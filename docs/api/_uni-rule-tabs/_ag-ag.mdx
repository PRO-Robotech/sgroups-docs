      -d '{
        "rules": [
          {
            "metadata": {
              "name": "allow-web-to-db",
              "namespace": "production",
              "labels": {
                "policy": "web-access"
              }
            },
            "spec": {
              "displayName": "Allow Web to DB",
              "comment": "Разрешить трафик от веб-серверов к БД",
              "action": "ALLOW",

                "traffic": "EGRESS"
              },
              "endpoints": {
                "local": {
                  "name": "web-servers",
                  "namespace": "production",
                  "type": "ADDRESS_GROUP"

    curl -X POST http://localhost:9006/v1/rules/upsert \\

      -d '{
        "rules": [
          {
            "metadata": {
              "name": "allow-web-to-internal-net",
              "namespace": "production"
            },
            "spec": {
              "action": "ALLOW",
              "session": {

              },
              "endpoints": {
                "local": {
                  "name": "web-servers",
